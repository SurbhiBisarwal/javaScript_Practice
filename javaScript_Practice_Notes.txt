function showPaymentOptions(){

function payChange(elem){
    let selOpt=elem.value
    const arr1=paymentOption.map(function(p1){
        return p1.type;
    });
    let dd1=makeDDfn("dd1",arr1,selOpt,'Select Payment Method','payChange');
    let json1=paymentOption.find(function(p1){
        return p1.type==selOpt;
    })
let dd2=makeDDfn("dd2",json1.providers,'','Select','');

    let div1=document.getElementById("payments");
    div1.innerHTML=dd1+dd2;
}

        function makeDDfn(id, data, selectedVal, valOnTop,onChangeFn) {
            let dataMap = data.map(function(opt) {
                if(opt === selectedVal) {
                    return '<option selected>' + opt + '</option>';
                } else {
                    return '<option>' + opt + '</option>';
                }
            });
            let first = '<option value=\'\' disabled>' + valOnTop + '</option>';
            if(selectedVal === '') {
                first = '<option value=\'\' selected disabled>' + valOnTop + '</option>';
            }
            let ddHtml = '<select id=\'' + id + '\'>';
                if(onChangeFn!=''){
                    ddHtml='<select id=\'' + id + '\' onchange=\''+onChangeFn+'(this)\'>';
                }
            ddHtml += first + dataMap.join('') + '</select>';
            return ddHtml;
        }

    <script>
        function calculateSum() {
            const num1 = document.getElementById("n1").value;
            const num2 = document.getElementById("n2").value;

            // Check if the inputs are valid numbers
            if (isNaN(num1) || isNaN(num2)) {
                document.getElementById("n").value = "Input correct numbers";
                return;
            }

            const number1 = parseFloat(num1) || 0;
            const number2 = parseFloat(num2) || 0;
            const sum = number1 + number2;
            
            document.getElementById("n").value = sum;
        }
    </script>
